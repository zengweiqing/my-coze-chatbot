    <script>
        // !!!重要：请替换成您自己的凭证 !!!
        const API_TOKEN = "pat_iqxnghDt4hkHwK1vdgsd7i1SNv9FeC5UUzShG3gD0VcdgNVH8gDdcyF9Dy27dtoF";
        const BOT_ID = "7569827924309295114";
        const API_URL = "https://api.coze.cn/v3/chat";

        async function sendMessage() {
            const userInput = document.getElementById('user-input');
            const sendButton = document.getElementById('send-button');
            const message = userInput.value.trim();

            if (!message) return;

            // 禁用输入和按钮，防止重复发送
            userInput.disabled = true;
            sendButton.disabled = true;

            // 显示用户消息
            displayMessage(message, 'user');
            userInput.value = '';

            try {
                const response = await fetch(API_URL, {
                    method: 'POST',
                    headers: {
                        'Content-Type': 'application/json',
                        'Authorization': `Bearer ${API_TOKEN}`
                    },
                    body: JSON.stringify({
                        bot_id: BOT_ID,
                        user_id: "anonymous_user_" + Date.now(), // 生成一个随机用户ID
                        stream: false, // 为简化流程，先使用非流式
                        query: message
                    })
                });

                const data = await response.json();
                console.log("API返回数据:", data);

                if (data.messages && data.messages.length > 0) {
                    // 找到最后一条助手消息
                    const lastBotMessage = data.messages.reverse().find(msg => msg.role === 'assistant');
                    if (lastBotMessage) {
                        displayMessage(lastBotMessage.content, 'bot');
                    } else {
                        throw new Error("助手未返回消息");
                    }
                } else {
                    throw new Error("API返回数据格式异常");
                }

            } catch (error) {
                console.error('调用API出错:', error);
                displayMessage("抱歉，网络或服务出现问题了，请稍后再试。", 'bot');
            } finally {
                // 重新启用输入和按钮
                userInput.disabled = false;
                sendButton.disabled = false;
                userInput.focus();
            }
        }

        function displayMessage(content, sender) {
            const chatMessages = document.getElementById('chat-messages');
            const messageDiv = document.createElement('div');
            messageDiv.classList.add('message', sender + '-message');

            const bubbleDiv = document.createElement('div');
            bubbleDiv.classList.add('message-bubble');
            bubbleDiv.textContent = content;

            messageDiv.appendChild(bubbleDiv);
            chatMessages.appendChild(messageDiv);

            // 滚动到底部
            chatMessages.scrollTop = chatMessages.scrollHeight;
        }

        // 按回车发送消息
        document.getElementById('user-input').addEventListener('keypress', function(e) {
            if (e.key === 'Enter') {
                sendMessage();
            }
        });
    </script>
